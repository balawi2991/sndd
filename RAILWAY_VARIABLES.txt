# ⚠️ اقرأ التعليمات قبل النسخ!

# ═══════════════════════════════════════════════════════════
# 1. JWT_SECRET (مطلوب - ولّده الآن!)
# ═══════════════════════════════════════════════════════════

# شغّل هذا الأمر في terminal:
# node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"

# ثم الصق الناتج هنا:
JWT_SECRET=<الصق_الناتج_هنا>

# مثال (لا تستخدم هذا!):
# JWT_SECRET=a1b2c3d4e5f6789012345678901234567890abcdef1234567890abcdef123456


# ═══════════════════════════════════════════════════════════
# 2. DEEPSEEK_API_KEY (مطلوب - احصل عليه من DeepSeek)
# ═══════════════════════════════════════════════════════════

# اذهب إلى: https://platform.deepseek.com
# سجل دخول → API Keys → Create API Key
# انسخ المفتاح والصقه هنا:

DEEPSEEK_API_KEY=<الصق_مفتاح_API_هنا>

# مثال (لا تستخدم هذا!):
# DEEPSEEK_API_KEY=sk-1234567890abcdef1234567890abcdef


# ═══════════════════════════════════════════════════════════
# 3. NODE_ENV (مطلوب - انسخ كما هو)
# ═══════════════════════════════════════════════════════════

NODE_ENV=production


# ═══════════════════════════════════════════════════════════
# ⚠️ لا تضف هذه المتغيرات - Railway يضيفها تلقائياً!
# ═══════════════════════════════════════════════════════════

# MONGODB_URI=<يضاف_تلقائياً_من_Railway>
# MONGO_URL=<يضاف_تلقائياً_من_Railway>
# PORT=<يضاف_تلقائياً_من_Railway>


# ═══════════════════════════════════════════════════════════
# ✅ ملخص: ما تحتاج إضافته في Railway
# ═══════════════════════════════════════════════════════════

# فقط هذه الثلاثة:
# 1. JWT_SECRET (ولّده بالأمر أعلاه)
# 2. DEEPSEEK_API_KEY (احصل عليه من platform.deepseek.com)
# 3. NODE_ENV=production (انسخ كما هو)


# ═══════════════════════════════════════════════════════════
# 📝 كيف تضيفها في Railway
# ═══════════════════════════════════════════════════════════

# 1. اذهب إلى Railway project
# 2. اختر service الخاص بك
# 3. اضغط "Variables" tab
# 4. اضغط "New Variable"
# 5. أضف كل متغير على حدة:
#    - Variable Name: JWT_SECRET
#    - Value: <الصق القيمة>
#    - اضغط "Add"
# 6. كرر للمتغيرات الأخرى
